CREATE TABLE TBL_MEMBER (
    ID VARCHAR2(30) PRIMARY KEY,
    PW VARCHAR2(30),
    NAME VARCHAR2(30)
);

CREATE TABLE PRODUCT_BOOK (
    CODE VARCHAR2(15) PRIMARY KEY,
    NAME VARCHAR2(50),
    PRICE INT,
    AUTHOR VARCHAR2(30),
    PUBLISHER VARCHAR2(30),
    INFO VARCHAR(200),
    IMAGE VARCHAR(100)
);

CREATE TABLE TBL_ADMIN
(
    ID   VARCHAR2(20) NOT NULL,
    PW   VARCHAR2(30) NOT NULL,
    NAME VARCHAR2(30),
    CONSTRAINT TBL_ADMIN_PK PRIMARY KEY (ID, PW)
);
